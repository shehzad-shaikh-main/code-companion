```C
// C program to illustrate 
// while loop
#include <stdio.h>
 
// Driver code
int main()
{
  // Initialization expression
  int i = 2; 
 
  // Test expression
  while(i < 10)
  {
    // loop body
    printf( "Hello World\n");   
    
    // update expression
    i++;
  } 
  
  return 0;
}
```
